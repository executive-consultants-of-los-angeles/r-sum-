[docs]    def get_sections(self, cv):
        """Get all Section objects for a document."""
        sections = [] 
        for section in list(
            Section.objects.filter(
                cv=cv
            ).order_by('id').values()
        ):
            if section.get('content') == u"<type 'list'>":
                ss = SubSection(section=self)
                section.update({
                    'content': ss.get_sub_section(
                        Section.objects.filter(
                            id=section.get('id')
                        )
                    ),
                })
            if section.get('content') == u"<type 'dict'>":
                ss = SubSection(section=self)
                section.update({
                    'content': ss.get_sub_section(
                        Section.objects.filter(
                            id=section.get('id')
                        )
                    ),
                })
            sections.append(section)
        return sections



[docs]    def save_section(self, cv, section, name):
        """Save one Section object."""
        if section is None:
            return None
        s_i = Section()
        s_i.cv = cv
        s_i.name = name

        if isinstance(section, str):
            s_i.content = section
            s_i.save()
        else:
            s_i.content = type(section)
            s_i.save()
            ss = SubSection()
            ss.save_sub_sections(section, s_i)

        return Section.objects.values()
